// server.js
// Minimal school-scoped accounts using SQLite on Replit
// Endpoints:
// POST /api/schools                { slug, name }
// POST /api/register               { email, password, displayName, schoolSlug }
// POST /api/login                  { email, password } -> returns { user, schools }
// GET  /api/schools/:slug/users    -> list users in that school

const express = require("express");
const bcrypt = require("bcryptjs");
const sqlite3 = require("sqlite3").verbose();
const path = require("path");

const app = express();
app.use(express.json());

// Create or open a persistent SQLite file
const db = new sqlite3.Database(path.join(__dirname, "db.sqlite"));

// Helpers
function run(db, sql, params = []) {
  return new Promise((resolve, reject) => {
    db.run(sql, params, function (err) {
      if (err) reject(err);
      else resolve(this);
    });
  });
}
function get(db, sql, params = []) {
  return new Promise((resolve, reject) => {
    db.get(sql, params, (err, row) => {
      if (err) reject(err);
      else resolve(row);
    });
  });
}
function all(db, sql, params = []) {
  return new Promise((resolve, reject) => {
    db.all(sql, params, (err, rows) => {
      if (err) reject(err);
      else resolve(rows);
    });
  });
}

// Initialize tables
const initSQL = `
PRAGMA foreign_keys = ON;

CREATE TABLE IF NOT EXISTS schools (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  slug TEXT UNIQUE NOT NULL,
  name TEXT NOT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS users (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  email TEXT UNIQUE NOT NULL,
  password_hash TEXT NOT NULL,
  display_name TEXT NOT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS school_memberships (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  school_id INTEGER NOT NULL,
  user_id INTEGER NOT NULL,
  role TEXT DEFAULT 'member',
  UNIQUE (school_id, user_id),
  FOREIGN KEY (school_id) REFERENCES schools(id) ON DELETE CASCADE,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE INDEX IF NOT EXISTS idx_memberships_school ON school_memberships (school_id);
`;

db.serialize(() => {
  db.exec(initSQL, (err) => {
    if (err) {
      console.error("DB init error:", err);
    } else {
      console.log("Database ready");
    }
  });
});

// Seed one example school if missing
async function ensureSeed() {
  const existing = await get(db, "SELECT id FROM schools WHERE slug = ?", [
    "princeton",
  ]);
  if (!existing) {
    await run(
      db,
      "INSERT INTO schools (slug, name) VALUES (?, ?)",
      ["princeton", "Princeton University"]
    );
    console.log("Seeded school princeton");
  }
}
ensureSeed().catch(console.error);

// Create a school
app.post("/api/schools", async (req, res) => {
  try {
    const { slug, name } = req.body;
    if (!slug || !name) return res.status(400).json({ error: "slug and name required" });
    await run(db, "INSERT INTO schools (slug, name) VALUES (?, ?)", [slug, name]);
    res.status(201).json({ ok: true });
  } catch (e) {
    res.status(400).json({ error: e.message });
  }
});

// Register user and attach to school in one go
app.post("/api/register", async (req, res) => {
  const { email, password, displayName, schoolSlug } = req.body;
  if (!email || !password || !displayName || !schoolSlug) {
    return res.status(400).json({ error: "email, password, displayName, schoolSlug required" });
  }

  try {
    const school = await get(db, "SELECT id FROM schools WHERE slug = ?", [schoolSlug]);
    if (!school) return res.status(404).json({ error: "school not found" });

    const hash = await bcrypt.hash(password, 12);

    // Use a transaction to ensure both inserts succeed together
    await run(db, "BEGIN");
    const userInsert = await run(
      db,
      "INSERT INTO users (email, password_hash, display_name) VALUES (?, ?, ?)",
      [email.toLowerCase(), hash, displayName]
    );
    const userId = userInsert.lastID;

    await run(
      db,
      "INSERT INTO school_memberships (school_id, user_id) VALUES (?, ?)",
      [school.id, userId]
    );
    await run(db, "COMMIT");

    res.status(201).json({ ok: true, userId });
  } catch (e) {
    try { await run(db, "ROLLBACK"); } catch {}
    res.status(400).json({ error: e.message });
  }
});

// Login and show which schools this user belongs to
app.post("/api/login", async (req, res) => {
  const { email, password } = req.body;
  if (!email || !password)
    return res.status(400).json({ error: "email and password required" });

  try {
    const user = await get(
      db,
      "SELECT id, email, password_hash, display_name FROM users WHERE email = ?",
      [email.toLowerCase()]
    );
    if (!user) return res.status(401).json({ error: "invalid credentials" });

    const ok = await bcrypt.compare(password, user.password_hash);
    if (!ok) return res.status(401).json({ error: "invalid credentials" });

    const schools = await all(
      db,
      `SELECT s.slug, s.name
       FROM schools s
       JOIN school_memberships sm ON sm.school_id = s.id
       WHERE sm.user_id = ?`,
      [user.id]
    );

    res.json({
      user: {
        id: user.id,
        email: user.email,
        displayName: user.display_name,
      },
      schools,
    });
  } catch (e) {
    res.status(400).json({ error: e.message });
  }
});

// List users in a school by slug
app.get("/api/schools/:slug/users", async (req, res) => {
  try {
    const { slug } = req.params;
    const school = await get(db, "SELECT id FROM schools WHERE slug = ?", [slug]);
    if (!school) return res.status(404).json({ error: "school not found" });

    const rows = await all(
      db,
      `SELECT u.id, u.display_name AS displayName, u.email
       FROM users u
       JOIN school_memberships sm ON sm.user_id = u.id
       WHERE sm.school_id = ?
       ORDER BY u.display_name`,
      [school.id]
    );
    res.json(rows);
  } catch (e) {
    res.status(400).json({ error: e.message });
  }
});

// Tiny homepage so you know it is running
app.get("/", (req, res) => {
  res.type("text").send(
`OK
Try:

  );
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log("Server running on", PORT));
